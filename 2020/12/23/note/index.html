## Temp < 70时显示“小”，否则显示大。Ozone > 100时显示红色，否则为蓝色
plot(x = aq$Temp, y = aq$Ozone, 
     pch = ifelse(aq$Temp < 70, "小", "大"), 
     col = ifelse(aq$Ozone > 100, "red", "blue"),
     cex = ifelse(aq$Solar.R > 200, 2, 1))

par(las = 1)
plot(x = aq$Temp, y = aq$Ozone, type = "h", 
          xlab = "Temperature", ylab = "Ozone")

# 读取张舒的数据，命名为“zs”
zs <- read.csv("D:\\R project\\mydata\\zs1.csv")

## 以x和y1画图，type=l为点连接成的折线,lty=3为点虚线，xy轴分别命名为x、y。隐藏坐标轴，y轴为120~160.
plot(zs$x, zs$y1, type = "l", lty = 3, 
     xlab = "x", ylab = "y", axes = FALSE,
     ylim = c(120, 160))

## x轴定义好大小为250, 500, 750, 1000, 1250, 1500，box()闭合
axis(1, at = c(0, 250, 500, 750, 1000,1250, 1500), labels = c(0, 250, 500, 750, 1000, 1250, 1500))
axis(2)
axis(3)
axis(4)
box()

## 添加曲线，x和y2的关系
lines(x = zs$x, y = zs$y2, col = "blue")

## 添加图例，
legend("topleft", lty = c(3, 1), col = c("black", "blue"), legend = c("y1", "y2"))
legend("bottomright", legend = "A")

## abline 画线，y=(a+bx),画一条y=x的线
abline(a = 0, b = 1)
## h：平行于横坐标轴，相对于y; v垂直于横坐标，相当于x.
abline(h = 140, col = "red")
abline(v = seq(0, 2000, by = 100), col = "red", lty = 2)

## 添加图表标题，
text(x = 750, y = 150, labels = "这是文字")

## 画多边形
polygon(x= c(200, 400, 600, 800), y = c(120, 150, 160, 140), col = "green")

## 多边形涂色， ## 数据表中第二第三列中的最大值下面部分涂为绿色，最小值下面部分涂为白色

plot(y$x, y$y2, type="n", ylim = c(120, 160), xlab = "x", ylab = "y")
polygon(x = x, y = y_green, col = "lightgreen", border = NA)
polygon(x = x, y = y_pink, col = "pink", border = NA)
polygon(x = x, y = y_white, col = "white", border = NA)
lines(y$x, y$y2)
